30  /**
31   * The SQL provider class that return the SQL template file path. <br>
32   * <b>IMPORTANT: This class required to use with mybatis 3.5.1+</b> and need to use with SQL provider annotation (such
33   * as {@link org.apache.ibatis.annotations.SelectProvider} as follow: <br>
34   * <br>
35   * 
36   * <pre>
37   * package com.example.mapper;
38   *
39   * public interface BaseMapper&lt;T&gt; {
40   *
41   *   &#64;Options(useGeneratedKeys = true, keyProperty = "id")
42   *   &#64;InsertProvider(type = TemplateFilePathProvider.class)
43   *   void insert(T entity);
44   *
45   *   &#64;UpdateProvider(type = TemplateFilePathProvider.class)
46   *   void update(T entity);
47   *
48   *   &#64;DeleteProvider(type = TemplateFilePathProvider.class)
49   *   void delete(T entity);
50   *
51   *   &#64;SelectProvider(type = TemplateFilePathProvider.class)
52   *   T findById(Integer id);
53   *
54   * }
55   * </pre>
56   *
57   * <pre>
58   * package com.example.mapper;
59   *
60   * public interface NameMapper extends BaseMapper {
61   *
62   *   &#64;SelectProvider(type = TemplateFilePathProvider.class)
63   *   List&lt;Name&gt; findByConditions(NameConditions conditions);
64   *
65   * }
66   * </pre>
67   * 
68   * @author Kazuki Shimizu
69   * @version 1.2.0
70   */

114   /**
115    * Provide an SQL scripting string(template file path).
116    *
117    * <br>
118    * By default implementation, a template file path resolve following format and priority order. If does not match all,
119    * it throw an exception that indicate not found a template file.
120    * <ul>
121    * <li>com/example/mapper/NameMapper/NameMapper-{methodName}-{databaseId}.ftl</li>
122    * <li>com/example/mapper/NameMapper/NameMapper-{methodName}.ftl (fallback using default database)</li>
123    * <li>com/example/mapper/BaseMapper/BaseMapper-{methodName}-{databaseId}.ftl (fallback using declaring class of
124    * method)</li>
125    * <li>com/example/mapper/BaseMapper/BaseMapper-{methodName}.ftl (fallback using declaring class of method and default
126    * database)</li>
127    * </ul>
128    * <br>
129    *
130    * @param context
131    *          a context of SQL provider
132    * @return an SQL scripting string(template file path)
133    */
134   @SuppressWarnings("unused")
135   public static String provideSql(ProviderContext context) {
136     return languageDriverConfig.getTemplateFile().getPathProvider().isCacheEnabled()
137         ? cache.computeIfAbsent(context, c -> providePath(c.getMapperType(), c.getMapperMethod(), c.getDatabaseId()))
138         : providePath(context.getMapperType(), context.getMapperMethod(), context.getDatabaseId());
139   }
140 
141   /**
142    * Clear cache.
143    */
144   public static void clearCache() {
145     cache.clear();
146   }
147 
148   static String providePath(Class<?> mapperType, Method mapperMethod, String databaseId) {
149     boolean fallbackDeclaringClass = mapperType != mapperMethod.getDeclaringClass();
150     boolean fallbackDatabase = databaseId != null;
151     String path = pathGenerator.generatePath(mapperType, mapperMethod, databaseId);
152     if (exists(path)) {
153       return path;
154     }
155     if (fallbackDatabase) {
156       path = pathGenerator.generatePath(mapperType, mapperMethod, null);
157       if (exists(path)) {
158         return path;
159       }
160     }
161     if (fallbackDeclaringClass) {
162       path = pathGenerator.generatePath(mapperMethod.getDeclaringClass(), mapperMethod, databaseId);
163       if (exists(path)) {
164         return path;
165       }
166       if (fallbackDatabase) {
167         path = pathGenerator.generatePath(mapperMethod.getDeclaringClass(), mapperMethod, null);
168         if (exists(path)) {
169           return path;
170         }
171       }
172     }
173     throw new IllegalStateException("The SQL template file not found. mapperType:[" + mapperType + "] mapperMethod:["
174         + mapperMethod + "] databaseId:[" + databaseId + "]");
175   }

177   private static String generateTemplatePath(Class<?> type, Method method, String databaseId) {
178     Package pkg = type.getPackage();
179     String packageName = pkg == null ? "" : pkg.getName();
180     String className = type.getName().substring(packageName.length() + (packageName.isEmpty() ? 0 : 1));
181 
182     PathProviderConfig pathProviderConfig = languageDriverConfig.getTemplateFile().getPathProvider();
183     StringBuilder path = new StringBuilder();
184     if (!pathProviderConfig.getPrefix().isEmpty()) {
185       path.append(pathProviderConfig.getPrefix());
186     }
187     if (pathProviderConfig.isIncludesPackagePath() && !packageName.isEmpty()) {
188       path.append(packageName.replace('.', '/')).append('/');
189     }
190     path.append(className);
191     if (pathProviderConfig.isSeparateDirectoryPerMapper()) {
192       path.append('/');
193       if (pathProviderConfig.isIncludesMapperNameWhenSeparateDirectory()) {
194         path.append(className).append('-');
195       }
196     } else {
197       path.append('-');
198     }
199     path.append(method.getName());
200     if (databaseId != null) {
201       path.append('-').append(databaseId);
202     }
203     path.append(".ftl");
204     return path.toString();
205   }
206 
207   private static boolean exists(String path) {
208     String basePath = languageDriverConfig.getTemplateFile().getBaseDir();
209     String actualPath = basePath.isEmpty() ? path : basePath + (basePath.endsWith("/") ? "" : "/") + path;
210     try {
211       return Resources.getResourceAsFile(actualPath).exists();
212     } catch (IOException e) {
213       return false;
214     }
215   }
