56    static {
57      Map<Class<?>, Function<String, Object>> converters = new HashMap<>();
58      converters.put(String.class, String::trim);
59      converters.put(boolean.class, v -> Boolean.valueOf(v.trim()));
60      converters.put(Object.class, v -> v);
61      TYPE_CONVERTERS = Collections.unmodifiableMap(converters);
62    }

350   /**
351    * Create an instance from specified properties.
352    *
353    * @param customProperties
354    *          custom configuration properties
355    * @return a configuration instance
356    * @see #newInstance()
357    */
358   public static FreeMarkerLanguageDriverConfig newInstance(Properties customProperties) {
359     FreeMarkerLanguageDriverConfig config = new FreeMarkerLanguageDriverConfig();
360     Properties properties = loadDefaultProperties();
361     Optional.ofNullable(customProperties).ifPresent(properties::putAll);
362     override(config, properties);
363     return config;
364   }
365 
366   /**
367    * Create an instance using specified customizer and override using a default properties file.
368    *
369    * @param customizer
370    *          baseline customizer
371    * @return a configuration instance
372    * @see #newInstance()
373    */
374   public static FreeMarkerLanguageDriverConfig newInstance(Consumer<FreeMarkerLanguageDriverConfig> customizer) {
375     FreeMarkerLanguageDriverConfig config = new FreeMarkerLanguageDriverConfig();
376     customizer.accept(config);
377     override(config, loadDefaultProperties());
378     return config;
379   }
380 
381   private static void override(FreeMarkerLanguageDriverConfig config, Properties properties) {
382     MetaObject metaObject = MetaObject.forObject(config, new DefaultObjectFactory(), new DefaultObjectWrapperFactory(),
383         new DefaultReflectorFactory());
384     properties.forEach((key, value) -> {
385       String propertyPath = WordUtils
386           .uncapitalize(WordUtils.capitalize(Objects.toString(key), '-').replaceAll("-", ""));
387       Optional.ofNullable(value).ifPresent(v -> {
388         Object convertedValue = TYPE_CONVERTERS.get(metaObject.getSetterType(propertyPath)).apply(value.toString());
389         metaObject.setValue(propertyPath, convertedValue);
390       });
391     });
392   }
393 
394   private static Properties loadDefaultProperties() {
395     return loadProperties(System.getProperty(PROPERTY_KEY_CONFIG_FILE, DEFAULT_PROPERTIES_FILE));
396   }
397 
398   private static Properties loadProperties(String resourcePath) {
399     Properties properties = new Properties();
400     InputStream in;
401     try {
402       in = Resources.getResourceAsStream(resourcePath);
403     } catch (IOException e) {
404       in = null;
405     }
406     if (in != null) {
407       Charset encoding = Optional.ofNullable(System.getProperty(PROPERTY_KEY_CONFIG_ENCODING)).map(Charset::forName)
408           .orElse(StandardCharsets.UTF_8);
409       try (InputStreamReader inReader = new InputStreamReader(in, encoding);
410           BufferedReader bufReader = new BufferedReader(inReader)) {
411         properties.load(bufReader);
412       } catch (IOException e) {
413         throw new IllegalStateException(e);
414       }
415     }
416     return properties;
417   }
