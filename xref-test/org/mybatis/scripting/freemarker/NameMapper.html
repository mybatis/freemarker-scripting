1   package org.mybatis.scripting.freemarker;
2   
3   import org.apache.ibatis.annotations.Lang;
4   import org.apache.ibatis.annotations.Param;
5   import org.apache.ibatis.annotations.Select;
6   
7   import java.util.List;
8   
9   /**
10   * Annotations-driven mapper for {@link org.mybatis.scripting.freemarker.FreeMarkerInAnnotationsTest}.
11   *
12   * @author elwood
13   */
14  public interface NameMapper {
15      /**
16       * Simple query that is loaded from template.
17       */
18      @Lang(FreeMarkerLanguageDriver.class)
19      @Select("getAllNames.ftl")
20      List<Name> getAllNames();
21  
22      /**
23       * Simple query with prepared statement parameter.
24       */
25      @Lang(FreeMarkerLanguageDriver.class)
26      @Select("findName.ftl")
27      Name findName(@Param("name") String name);
28  
29      /**
30       * If any whitespace found inside @Select text, it is interpreted as inline script, not template name.
31       * It is convenient to avoid creating templates when script is really small.
32       */
33      @Lang(FreeMarkerLanguageDriver.class)
34      @Select("select * from names where id in (${ids?join(',')})")
35      List<Name> findNamesByIds(@Param("ids") List<Integer> ids);
36  
37      /**
38       * There are no @Param annotation on argument. This means NameParam instance
39       * will be passed into driver as is, not as Map entry. So, we need to support this case.
40       * Because in driver we need to add some another properties into template model,
41       * and NameParam is not Map, we are need to wrap passed parameter object into
42       * {@link org.mybatis.scripting.freemarker.ParamObjectAdapter} before processing template.
43       */
44      @Lang(FreeMarkerLanguageDriver.class)
45      @Select("select * from names where id = <@p name='id'/> and id = ${id}")
46      Name find(NameParam nameParam);
47  
48      /**
49       * This query is to demonstrate MyBatis odd behaviour when using String as parameter
50       * and can use properties that not exist. Both props will be use provided `name` parameter value.
51       * Goal is to write FreeMarker lang plugin to support this behaviour too (although it is confusing one).
52       */
53      @Select("select * from names" +
54              " where firstName = #{noSuchPropertyOnString}" +
55              " or firstName = #{oneMoreUnexistingProperty}")
56      List<Name> getNamesOddBehaviourStdLang(String name);
57  
58      /**
59       * This query is to demonstrate that FreeMarker does not break the compatibility with this behaviour.
60       */
61      @Lang(FreeMarkerLanguageDriver.class)
62      @Select("select * from names" +
63              " where firstName = <@p name='noSuchPropertyOnString'/>" +
64              " or firstName = <@p name='oneMoreUnexistingProperty'/>")
65      List<Name> getNamesOddBehaviourFreeMarkerLang(String name);
66  }
